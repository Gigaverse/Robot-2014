199 bool Command::Run()
200 {
201  if (!m_runWhenDisabled && m_parent == NULL && DriverStation::GetInstance()->IsDisabled())
202  Cancel();
203 
204  if (IsCanceled())
205  return false;
206 
207  if (!m_initialized)
208  {
209  m_initialized = true;
210  StartTiming();
211  _Initialize();
212  Initialize();
213  }
214  _Execute();
215  Execute();
216  return !IsFinished();
217 }
