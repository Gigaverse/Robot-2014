21 static void CompressorChecker(Compressor *c)
22 {
23  while (1)
24  {
25  if (c->Enabled())
26  {
27  c->SetRelayValue( c->GetPressureSwitchValue() == 0 ? Relay::kOn : Relay::kOff );
28  }
29  else
30  {
31  c->SetRelayValue(Relay::kOff);
32  }
33  Wait(0.5);
34  }
35 }
